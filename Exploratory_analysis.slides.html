#Univariate with Horizental Bar
#Horizental bar chart - Bar charts are arguably the simplest data visualization. They map categories to numbers
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
trips = pd.read_csv("2017-fordgobike-tripdata.csv", index_col=0)
Male = mpatches.Patch(color='blue', label='Male')
Female = mpatches.Patch(color='orange', label='Female')
Others = mpatches.Patch(color='green', label='Others')

ax=(trips['member_gender']).value_counts().plot.barh()
ax.set_ylabel('Gender', fontsize=13, color='blue')
ax.set_xlabel('Trip Counts', fontsize=13, color='blue')
ax.set_title('Fordgobike trip count by gender',  fontsize=13, color='blue')
total=trips['member_gender'].value_counts()

#Line chart - A line chart can pass over any number of many individual values, 
#making it the tool of first choice for distributions with many unique values or categories.

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
loans = pd.read_csv("prosperLoanData.csv", index_col=0)
creditlineyear = np.arange(loans['FirstRecordedCreditLine'].shape[0])
yields = loans['EstimatedEffectiveYield']
loss = loans['EstimatedLoss']
returns = loans['EstimatedReturn']

plt.plot(creditlineyear, returns, color='blue')

#Histogram - A histogram looks, trivially, like a bar plot. And it basically is! In fact,
#a histogram is special kind of bar plot that splits your data into even intervals and 
#displays how many rows are in each interval with bars. 
#The only analytical difference is that instead of each bar representing a single value, it represents a range of values.

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
loans = pd.read_csv("prosperLoanData.csv", index_col=0)

plt.title('RevolvingCreditBalance less than 10K',fontsize=13, color='blue')
plt.legend(loc='upper right',bbox_to_anchor=(1, 1))

#scatter plots

#This scatterplot uses three visual variables. The horizontal position (x-value) tracks the Value of the player 
#(how well they are paid). The vertical position (y-value) tracks the Overall score of the player across all attributes. 
#And the color (the hue parameter) tracks which of the three categories of interest the player the point represents is in.

import pandas as pd
pd.set_option('max_columns', None)
df = pd.read_csv("CompleteDataset.csv", index_col=0)

import re
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

footballers = df.copy()
footballers['Unit'] = df['Value'].str[-1]
footballers['Value (M)'] = np.where(footballers['Unit'] == '0', 0, 
                                    footballers['Value'].str[1:-1].replace(r'[a-zA-Z]',''))
footballers['Value (M)'] = footballers['Value (M)'].astype(float)
footballers['Value (M)'] = np.where(footballers['Unit'] == 'M', 
                                    footballers['Value (M)'], 
                                    footballers['Value (M)']/1000)
footballers = footballers.assign(Value=footballers['Value (M)'],
                                 Position=footballers['Preferred Positions'].str.split().str[0])

g=sns.lmplot(x='Value', y='Overall', markers=['o', 'x', '*'], hue='Position',
           data=footballers.loc[footballers['Position'].isin(['ST', 'RW', 'LW'])],
           fit_reg=False
          )
plt.title('Overall Value and Preferred position', fontsize=15, color='blue')
sns.set(font_scale=1.5)
